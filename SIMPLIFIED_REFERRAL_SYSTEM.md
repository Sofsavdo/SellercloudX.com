```sql
CREATE TABLE partner_subscriptions (

  start_date TIMESTAMP,

CREATE TABLE referral_bonuses (
  id TEXT PRIMARY KEY,
  referrer_id TEXT NOT NULL,
  referred_id TEXT NOT NULL,
  subscription_id TEXT NOT NULL,
  bonus_percent REAL NOT NULL, -- 5, 8, or 10
  bonus_amount REAL NOT NULL,
  status TEXT DEFAULT 'pending',
  paid_at TIMESTAMP,
  created_at TIMESTAMP
);
```

### Business Logic

```typescript
function canRefer(partner: Partner): boolean {
  const subscription = partner.currentSubscription;
  return subscription.duration >= 3 && subscription.status === 'active';
}

function calculateBonus(tier: string, duration: number) {
  if (duration === 1) return { bonus: 0, discount: 3 };
  if (duration === 3) return { bonus: 5, discount: 5 };
  if (duration === 6) return { bonus: 8, discount: 5 };
  if (duration === 12) return { bonus: 10, discount: 5 };
}

function processReferral(referrerId: string, referredId: string, subscriptionId: string) {
  const subscription = getSubscription(subscriptionId);
  const bonus = calculateBonus(subscription.tier, subscription.duration);
  
  if (bonus.bonus > 0) {
    createBonus({
      referrerId,
      referredId,
      subscriptionId,
      bonusPercent: bonus.bonus,
      bonusAmount: subscription.amount * (bonus.bonus / 100),
      status: 'pending'
    });
  }
}
```
